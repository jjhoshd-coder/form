    اطلب الآن عبر واتساب let cart = []; // تحديث عداد السلة function updateCartCount() { const cartCount = document.querySelector(".cart span"); cartCount.textContent = cart.length; } // تحديث عرض السلة function renderCart() { const cartList = document.getElementById("cart-items-list"); cartList.innerHTML = ""; cart.forEach((item, index) => { const li = document.createElement("li"); li.innerHTML = ` ${item.name} - $${item.price} x `; cartList.appendChild(li); }); // تعديل الكمية document.querySelectorAll(".cart-qty").forEach(input => { input.addEventListener("change", (e) => { const i = e.target.getAttribute("data-index"); cart[i].quantity = parseInt(e.target.value); updateCartCount(); }); }); // حذف المنتج document.querySelectorAll(".remove-item").forEach(btn => { btn.addEventListener("click", (e) => { const i = e.target.getAttribute("data-index"); cart.splice(i, 1); updateCartCount(); renderCart(); }); }); } // إضافة المنتجات للسلة document.querySelectorAll(".add-to-cart").forEach(button => { button.addEventListener("click", (e) => { e.preventDefault(); const productBox = button.closest(".product-box"); const title = productBox.querySelector(".product-text-title").textContent; const priceText = productBox.querySelector("span").textContent; const price = parseFloat(priceText.replace("$", "")); // إذا المنتج موجود نزود الكمية بدل الإضافة مرة جديدة const existing = cart.find(item => item.name === title); if (existing) { existing.quantity++; } else { cart.push({ name: title, price: price, quantity: 1 }); } updateCartCount(); renderCart(); alert(`${title} تم إضافته للسلة!`); }); }); // ربط السلة بالفورم const form = document.querySelector(".form-container form"); form.addEventListener("submit", function() { const cartItemsInput = document.getElementById("cart-items-input"); cartItemsInput.value = JSON.stringify(cart); });
